import warnings
warnings.simplefilter(action='ignore', category=DeprecationWarning)

ty='BLCA BRCA CESC CRC ESCA GBM HNSC KIRC KIRP LGG LIHC LUAD LUSC OV PAAD PRAD SKCM STAD THCA UCEC'.split()
# lig rec
lr=pd.read_table('data/1370_pairs.txt',names=['lr'])
lr['l']=lr.lr.apply(lambda x:x.split('_')[0])
lr['r']=lr.lr.apply(lambda x:x.split('_')[1])

# filter median rec and lig exp in >=25% samples, 604 receptors
e=pd.read_csv('data/pct_samp_fpkm1.csv.gz',index_col=0).median(axis=1)

e=e[e>=.25].index.intersection(set(lr.r).union(set(lr.l)))

lr=lr[lr.r.isin(e)]
lr=lr[lr.l.isin(e)]

#normal
n=2**pd.read_csv('data/normal_median_logx+1.csv.gz',index_col=0) -1

#scores
s=pd.DataFrame(index=lr.lr)

for t in ty:	
	d=2**pd.read_table('data/'+t+'_log_deconv_nnls_cnv.gz', index_col=0, sep=' ')-1
	
	cc = d.cancer.ix[lr.l].values*d.cancer.ix[lr.r].values	
	cs = d.cancer.ix[lr.l].values*d.stroma.ix[lr.r].values

	ss = d.stroma.ix[lr.l].values*d.stroma.ix[lr.r].values
	
	nn=n[t].ix[lr.l].values*n[t].ix[lr.r].values
	
	#sum=
	s[t+'_cl']=d.cancer.ix[lr.l].values
	s[t+'_cr']=d.cancer.ix[lr.r].values
	s[t+'_sl']=d.stroma.ix[lr.l].values
	s[t+'_sr']=d.stroma.ix[lr.r].values
	s[t+'_nl']=n[t].ix[lr.l].values
	s[t+'_nr']=n[t].ix[lr.r].values

	s[t+'_cs_s']=cs
	s[t+'_sc_s']=sc
	s[t+'_ss_s']=ss
	s[t+'_nn_s']=nn
	s[t+'_sum']=cc+cs+sc+ss+nn
	
	s[t+'_cc']=cc*100/s[t+'_sum']
	s[t+'_cs']=cs*100/s[t+'_sum']
	s[t+'_sc']=sc*100/s[t+'_sum']
